67class BlobIOStream : public IOStream {
68public:
73 BlobIOStream(BlobIOSystem *creator, const std::string &file, size_t initial = 4096) :
74 buffer(),
75 cur_size(),
76 file_size(),
77 cursor(),
78 initial(initial),
79 file(file),
80 creator(creator) {
81 // empty
82 }
83
85 ~BlobIOStream() override;
86
87public:
88 // -------------------------------------------------------------------
89 aiExportDataBlob *GetBlob() {
90 aiExportDataBlob *blob = new aiExportDataBlob();
91 blob->size = file_size;
92 blob->data = buffer;
93
94 buffer = nullptr;
95
96 return blob;
97 }
98
99 // -------------------------------------------------------------------
100 size_t Read(void *, size_t, size_t) override {
101 return 0;
102 }
103
104 // -------------------------------------------------------------------
105 size_t Write(const void *pvBuffer, size_t pSize, size_t pCount) override {
106 pSize *= pCount;
107 if (cursor + pSize > cur_size) {
108 Grow(cursor + pSize);
109 }
110
111 memcpy(buffer + cursor, pvBuffer, pSize);
112 cursor += pSize;
113
114 file_size = std::max(file_size, cursor);
115 return pCount;
116 }
117
118 // -------------------------------------------------------------------
119 aiReturn Seek(size_t pOffset, aiOrigin pOrigin) override {
120 switch (pOrigin) {
121 case aiOrigin_CUR:
122 cursor += pOffset;
123 break;
124
125 case aiOrigin_END:
126 cursor = file_size - pOffset;
127 break;
128
129 case aiOrigin_SET:
130 cursor = pOffset;
131 break;
132
133 default:
134 return AI_FAILURE;
135 }
136
137 if (cursor > file_size) {
138 Grow(cursor);
139 }
140
141 file_size = std::max(cursor, file_size);
142
143 return AI_SUCCESS;
144 }
145
146 // -------------------------------------------------------------------
147 size_t Tell() const override {
148 return cursor;
149 }
150
151 // -------------------------------------------------------------------
152 size_t FileSize() const override {
153 return file_size;
154 }
155
156 // -------------------------------------------------------------------
157 void Flush() override {
158 // ignore
159 }
160
161private:
162 // -------------------------------------------------------------------
163 void Grow(size_t need = 0) {
164 // 1.5 and phi are very heap-friendly growth factors (the first
165 // allows for frequent re-use of heap blocks, the second
166 // forms a fibonacci sequence with similar characteristics -
167 // since this heavily depends on the heap implementation
168 // and other factors as well, i'll just go with 1.5 since
169 // it is quicker to compute).
170 size_t new_size = std::max(initial, std::max(need, cur_size + (cur_size >> 1)));
171
172 const uint8_t *const old = buffer;
173 buffer = new uint8_t[new_size];
174
175 if (old) {
176 memcpy(buffer, old, cur_size);
177 delete[] old;
178 }
179
180 cur_size = new_size;
181 }
182
183private:
184 uint8_t *buffer;
185 size_t cur_size, file_size, cursor, initial;
186
187 const std::string file;
188 BlobIOSystem *const creator;
189};

196class BlobIOSystem : public IOSystem {
197
198 friend class BlobIOStream;
199 typedef std::pair<std::string, aiExportDataBlob *> BlobEntry;
200
201
202public:
204 BlobIOSystem() :
205 baseName{AI_BLOBIO_MAGIC} {
206 }
207
210 BlobIOSystem(const std::string &baseName) :
211 baseName(baseName) {
212 // empty
213 }
214
215 ~BlobIOSystem() override {
216 for (BlobEntry &blobby : blobs) {
217 delete blobby.second;
218 }
219 }
220
221public:
222 // -------------------------------------------------------------------
223 const char *GetMagicFileName() const {
224 return baseName.c_str();
225 }
226
227 // -------------------------------------------------------------------
228 aiExportDataBlob *GetBlobChain() {
229 const auto magicName = std::string(this->GetMagicFileName());
230 const bool hasBaseName = baseName != AI_BLOBIO_MAGIC;
231
232 // one must be the master
233 aiExportDataBlob *master = nullptr, *cur;
234
235 for (const BlobEntry &blobby : blobs) {
236 if (blobby.first == magicName) {
237 master = blobby.second;
238 master->name.Set(hasBaseName ? blobby.first : "");
239 break;
240 }
241 }
242
243 if (!master) {
244 ASSIMP_LOG_ERROR("BlobIOSystem: no data written or master file was not closed properly.");
245 return nullptr;
246 }
247
248 cur = master;
249
250 for (const BlobEntry &blobby : blobs) {
251 if (blobby.second == master) {
252 continue;
253 }
254
255 cur->next = blobby.second;
256 cur = cur->next;
257
258 if (hasBaseName) {
259 cur->name.Set(blobby.first);
260 } else {
261 // extract the file extension from the file written
262 const std::string::size_type s = blobby.first.find_first_of('.');
263 cur->name.Set(s == std::string::npos ? blobby.first : blobby.first.substr(s + 1));
264 }
265 }
266
267 // give up blob ownership
268 blobs.clear();
269 return master;
270 }
271
272public:
273 // -------------------------------------------------------------------
274 bool Exists(const char *pFile) const override {
275 return created.find(std::string(pFile)) != created.end();
276 }
277
278 // -------------------------------------------------------------------
279 char getOsSeparator() const override {
280 return '/';
281 }
282
283 // -------------------------------------------------------------------
284 IOStream *Open(const char *pFile, const char *pMode) override {
285 if (pMode[0] != 'w') {
286 return nullptr;
287 }
288
289 created.insert(std::string(pFile));
290 return new BlobIOStream(this, std::string(pFile));
291 }
292
293 // -------------------------------------------------------------------
294 void Close(IOStream *pFile) override {
295 delete pFile;
296 }
297
298private:
299 // -------------------------------------------------------------------
300 void OnDestruct(const std::string &filename, BlobIOStream *child) {
301 // we don't know in which the files are closed, so we
302 // can't reliably say that the first must be the master
303 // file ...
304 blobs.emplace_back(filename, child->GetBlob());
305 }
306
307private:
308 std::string baseName;
309 std::set<std::string> created;
310 std::vector<BlobEntry> blobs;
311};
