271template <class TNodeType>
272inline TNodeType *TXmlParser<TNodeType>::findNode(const std::string &name) {
273 if (name.empty()) {
274 return nullptr;
275 }
276
277 if (nullptr == mDoc) {
278 return nullptr;
279 }
280
281 find_node_by_name_predicate predicate(name);
282 mCurrent = mDoc->find_node(std::move(predicate));
283 if (mCurrent.empty()) {
284 return nullptr;
285 }
286
287 return &mCurrent;

295template <class TNodeType>
296bool TXmlParser<TNodeType>::parse(IOStream *stream) {
297 if (hasRoot()) {
298 clear();
299 }
300
301 if (nullptr == stream) {
302 ASSIMP_LOG_DEBUG("Stream is nullptr.");
303 return false;
304 }
305
306 const size_t len = stream->FileSize();
307 mData.resize(len + 1);
308 memset(&mData[0], '\0', len + 1);
309 stream->Read(&mData[0], 1, len);
310
311 mDoc = new pugi::xml_document();
312 // load_string assumes native encoding (aka always utf-8 per build options)
313 //pugi::xml_parse_result parse_result = mDoc->load_string(&mData[0], pugi::parse_full);
314 pugi::xml_parse_result parse_result = mDoc->load_buffer(&mData[0], mData.size(), pugi::parse_full);
315 if (parse_result.status == pugi::status_ok) {
316 return true;
317 }
318
319 ASSIMP_LOG_DEBUG("Error while parse xml.", std::string(parse_result.description()), " @ ", parse_result.offset);
320
321 return false;
322}

510class XmlNodeIterator {
511public:
513 enum IterationMode {
514 PreOrderMode,
515 PostOrderMode
516 };
520 explicit XmlNodeIterator(XmlNode &parent, IterationMode mode) :
521 mParent(parent),
522 mNodes(),
523 mIndex(0) {
524 if (mode == PreOrderMode) {
525 collectChildrenPreOrder(parent);
526 } else {
527 collectChildrenPostOrder(parent);
528 }
529 }
530
532 ~XmlNodeIterator() = default;
533
536 void collectChildrenPreOrder(XmlNode &node) {
537 if (node != mParent && node.type() == pugi::node_element) {
538 mNodes.push_back(node);
539 }
540 for (XmlNode currentNode : node.children()) {
541 collectChildrenPreOrder(currentNode);
542 }
543 }
544
547 void collectChildrenPostOrder(XmlNode &node) {
548 for (XmlNode currentNode = node.first_child(); currentNode; currentNode = currentNode.next_sibling()) {
549 collectChildrenPostOrder(currentNode);
550 }
551 if (node != mParent) {
552 mNodes.push_back(node);
553 }
554 }
555
559 bool getNext(XmlNode &next) {
560 if (mIndex == mNodes.size()) {
561 return false;
562 }
563
564 next = mNodes[mIndex];
565 ++mIndex;
566
567 return true;
568 }
569
572 size_t size() const {
573 return mNodes.size();
574 }
575
578 bool isEmpty() const {
579 return mNodes.empty();
580 }
581
583 void clear() {
584 if (mNodes.empty()) {
585 return;
586 }
587
588 mNodes.clear();
589 mIndex = 0;
590 }
591
592private:
593 XmlNode &mParent;
594 std::vector<XmlNode> mNodes;
595 size_t mIndex;
596};
