66 // std::plus-family operates on operands with identical types - we need to
67 // support all the (vectype op float) combinations in vector maths.
68 // Providing T(float) would open the way to endless implicit conversions.
70 namespace Intern {
71 template <typename T0, typename T1, typename TRES = T0> struct plus {
72 TRES operator() (const T0& t0, const T1& t1) const {
73 return t0+t1;
74 }
75 };
76 template <typename T0, typename T1, typename TRES = T0> struct minus {
77 TRES operator() (const T0& t0, const T1& t1) const {
78 return t0-t1;
79 }
80 };
81 template <typename T0, typename T1, typename TRES = T0> struct multiplies {
82 TRES operator() (const T0& t0, const T1& t1) const {
83 return t0*t1;
84 }
85 };
86 template <typename T0, typename T1, typename TRES = T0> struct divides {
87 TRES operator() (const T0& t0, const T1& t1) const {
88 return t0/t1;
89 }
90 };
91 }

213 template <template <typename t> class op> static Vertex BinaryOp(const Vertex& v0, const Vertex& v1) {
214 // this is a heavy task for the compiler to optimize ... *pray*
215
216 Vertex res;
217 res.position = op<aiVector3D>()(v0.position,v1.position);
218 res.normal = op<aiVector3D>()(v0.normal,v1.normal);
219 res.tangent = op<aiVector3D>()(v0.tangent,v1.tangent);
220 res.bitangent = op<aiVector3D>()(v0.bitangent,v1.bitangent);
221
222 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_TEXTURECOORDS; ++i) {
223 res.texcoords[i] = op<aiVector3D>()(v0.texcoords[i],v1.texcoords[i]);
224 }
225 for (unsigned int i = 0; i < AI_MAX_NUMBER_OF_COLOR_SETS; ++i) {
226 res.colors[i] = op<aiColor4D>()(v0.colors[i],v1.colors[i]);
227 }
228 return res;
229 }
