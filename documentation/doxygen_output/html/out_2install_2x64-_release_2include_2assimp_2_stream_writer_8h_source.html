215 void PutString(const aiString& s)
216 {
217 // as Put(T f) below
218 if (cursor + s.length >= buffer.size()) {
219 buffer.resize(cursor + s.length);
220 }
221 void* dest = &buffer[cursor];

259 // ---------------------------------------------------------------------
261 template <typename T>
262 void Put(T f) {
263 Intern :: Getter<SwapEndianness,T,RuntimeSwitch>() (&f, le);
264
265 if (cursor + sizeof(T) >= buffer.size()) {
266 buffer.resize(cursor + sizeof(T));
267 }
268
269 void* dest = &buffer[cursor];
270
271 // reinterpret_cast + assignment breaks strict aliasing rules
272 // and generally causes trouble on platforms such as ARM that
273 // do not silently ignore alignment faults.
274 ::memcpy(dest, &f, sizeof(T));
275 cursor += sizeof(T);
276 }
277

284 std::size_t cursor;
