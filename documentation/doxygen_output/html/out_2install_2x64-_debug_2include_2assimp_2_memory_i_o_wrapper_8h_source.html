66class MemoryIOStream : public IOStream {
67public:
68 MemoryIOStream (const uint8_t* buff, size_t len, bool own = false) :
69 buffer (buff),
70 length(len),
71 pos(static_cast<size_t>(0)),
72 own(own) {
73 // empty
74 }
75
76 ~MemoryIOStream() override {
77 if(own) {
78 delete[] buffer;
79 }
80 }
81
82 size_t Read(void* pvBuffer, size_t pSize, size_t pCount) override {
83 ai_assert(nullptr != pvBuffer);
84 ai_assert(0 != pSize);
85
86 const size_t cnt = std::min( pCount, (length-pos) / pSize);
87 const size_t ofs = pSize * cnt;
88
89 ::memcpy(pvBuffer,buffer+pos,ofs);
90 pos += ofs;
91
92 return cnt;
93 }
94
95 size_t Write(const void*, size_t, size_t ) override {
96 ai_assert(false); // won't be needed
97 return 0;
98 }
99
100 aiReturn Seek(size_t pOffset, aiOrigin pOrigin) override {
101 if (aiOrigin_SET == pOrigin) {
102 if (pOffset > length) {
103 return AI_FAILURE;
104 }
105 pos = pOffset;
106 } else if (aiOrigin_END == pOrigin) {
107 if (pOffset > length) {
108 return AI_FAILURE;
109 }
110 pos = length-pOffset;
111 } else {
112 if (pOffset+pos > length) {
113 return AI_FAILURE;
114 }
115 pos += pOffset;
116 }
117 return AI_SUCCESS;
118 }
119
120 size_t Tell() const override {
121 return pos;
122 }
123
124 size_t FileSize() const override {
125 return length;
126 }
127
128 void Flush() override{
129 ai_assert(false); // won't be needed
130 }
131
132private:
133 const uint8_t* buffer;
134 size_t length,pos;
135 bool own;
136};

140class MemoryIOSystem : public IOSystem {
141public:
143 MemoryIOSystem(const uint8_t* buff, size_t len, IOSystem* io) : buffer(buff), length(len), existing_io(io) {
144 // empty
145 }
146
148 ~MemoryIOSystem() override = default;
149
150 // -------------------------------------------------------------------
152 bool Exists(const char* pFile) const override {
153 if (0 == strncmp( pFile, AI_MEMORYIO_MAGIC_FILENAME, AI_MEMORYIO_MAGIC_FILENAME_LENGTH ) ) {
154 return true;
155 }
156 return existing_io ? existing_io->Exists(pFile) : false;
157 }
158
159 // -------------------------------------------------------------------
161 char getOsSeparator() const override {
162 return existing_io ? existing_io->getOsSeparator()
163 : '/'; // why not? it doesn't care
164 }
165
166 // -------------------------------------------------------------------
168 IOStream* Open(const char* pFile, const char* pMode = "rb") override {
169 if ( 0 == strncmp( pFile, AI_MEMORYIO_MAGIC_FILENAME, AI_MEMORYIO_MAGIC_FILENAME_LENGTH ) ) {
170 created_streams.emplace_back(new MemoryIOStream(buffer, length));
171 return created_streams.back();
172 }
173 return existing_io ? existing_io->Open(pFile, pMode) : nullptr;
174 }
175
176 // -------------------------------------------------------------------
178 void Close( IOStream* pFile) override {
179 auto it = std::find(created_streams.begin(), created_streams.end(), pFile);
180 if (it != created_streams.end()) {
181 delete pFile;
182 created_streams.erase(it);
183 } else if (existing_io) {
184 existing_io->Close(pFile);
185 }
186 }
187
188 // -------------------------------------------------------------------
190 bool ComparePaths(const char* one, const char* second) const override {
191 return existing_io ? existing_io->ComparePaths(one, second) : false;
192 }
193
195 bool PushDirectory( const std::string &path ) override {
196 return existing_io ? existing_io->PushDirectory(path) : false;
197 }
198
200 const std::string &CurrentDirectory() const override {
201 static std::string empty;
202 return existing_io ? existing_io->CurrentDirectory() : empty;
203 }
204
206 size_t StackSize() const override {
207 return existing_io ? existing_io->StackSize() : 0;
208 }
209
211 bool PopDirectory() override {
212 return existing_io ? existing_io->PopDirectory() : false;
213 }
214
216 bool CreateDirectory( const std::string &path ) override {
217 return existing_io ? existing_io->CreateDirectory(path) : false;
218 }
219
221 bool ChangeDirectory( const std::string &path ) override {
222 return existing_io ? existing_io->ChangeDirectory(path) : false;
223 }
224
226 bool DeleteFile( const std::string &file ) override {
227 return existing_io ? existing_io->DeleteFile(file) : false;
228 }
229
230private:
231 const uint8_t* buffer;
232 size_t length;
233 IOSystem* existing_io;
234 std::vector<IOStream*> created_streams;
235};
