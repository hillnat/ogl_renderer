80inline unsigned int ASSIMP_itoa10(char *out, unsigned int max, int32_t number) {
81 ai_assert(nullptr != out);
82
83 // write the unary minus to indicate we have a negative number
84 unsigned int written = 1u;
85 if (number < 0 && written < max) {
86 *out++ = '-';
87 ++written;
88 number = -number;
89 }
90
91 // We begin with the largest number that is not zero.
92 int32_t cur = 1000000000; // 2147483648
93 bool mustPrint = false;
94 while (written < max) {
95
96 const unsigned int digit = number / cur;
97 if (mustPrint || digit > 0 || 1 == cur) {
98 // print all future zero's from now
99 mustPrint = true;
100
101 *out++ = '0' + static_cast<char>(digit);
102
103 ++written;
104 number -= digit * cur;
105 if (1 == cur) {
106 break;
107 }
108 }
109 cur /= 10;
110 }
111
112 // append a terminal zero
113 *out++ = '\0';
114 return written - 1;
115}

138inline int ASSIMP_stricmp(const char *s1, const char *s2) {
139 ai_assert(nullptr != s1);
140 ai_assert(nullptr != s2);
141
142#if (defined _MSC_VER)
143
144 return ::_stricmp(s1, s2);
145#else
146 char c1, c2;
147 do {
148 c1 = tolower((unsigned char)*(s1++));
149 c2 = tolower((unsigned char)*(s2++));
150 } while (c1 && (c1 == c2));
151 return c1 - c2;
152#endif
153}

179inline int ASSIMP_strincmp(const char *s1, const char *s2, unsigned int n) {
180 ai_assert(nullptr != s1);
181 ai_assert(nullptr != s2);
182 if (!n) {
183 return 0;
184 }
185
186#if (defined _MSC_VER)
187
188 return ::_strnicmp(s1, s2, n);
189
190#elif defined(__GNUC__)
191
192 return ::strncasecmp(s1, s2, n);
193
194#else
195 char c1, c2;
196 unsigned int p = 0;
197 do {
198 if (p++ >= n) return 0;
199 c1 = tolower((unsigned char)*(s1++));
200 c2 = tolower((unsigned char)*(s2++));
201 } while (c1 && (c1 == c2));
202
203 return c1 - c2;
204#endif
205}

213 unsigned int res = 1;
