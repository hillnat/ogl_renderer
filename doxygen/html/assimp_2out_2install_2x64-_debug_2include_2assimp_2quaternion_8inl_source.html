64aiQuaterniont<TReal> operator * (const aiMatrix4x4t<TReal>& pMatrix, const aiQuaterniont<TReal>& pQuaternion) {
65 aiQuaterniont<TReal> res;
66 res.x = pMatrix.a1 * pQuaternion.x + pMatrix.a2 * pQuaternion.y + pMatrix.a3 * pQuaternion.z + pMatrix.a4 * pQuaternion.w;
67 res.y = pMatrix.b1 * pQuaternion.x + pMatrix.b2 * pQuaternion.y + pMatrix.b3 * pQuaternion.z + pMatrix.b4 * pQuaternion.w;
68 res.z = pMatrix.c1 * pQuaternion.x + pMatrix.c2 * pQuaternion.y + pMatrix.c3 * pQuaternion.z + pMatrix.c4 * pQuaternion.w;
69 res.w = pMatrix.d1 * pQuaternion.x + pMatrix.d2 * pQuaternion.y + pMatrix.d3 * pQuaternion.z + pMatrix.d4 * pQuaternion.w;
70 return res;
71}

104// ---------------------------------------------------------------------------
105// Constructs a quaternion from a rotation matrix
106template<typename TReal>
107inline aiQuaterniont<TReal>::aiQuaterniont( const aiMatrix3x3t<TReal> &pRotMatrix)
108{
109 TReal t = pRotMatrix.a1 + pRotMatrix.b2 + pRotMatrix.c3;
110
111 // large enough
112 if( t > static_cast<TReal>(0))
113 {
114 TReal s = std::sqrt(1 + t) * static_cast<TReal>(2.0);
115 x = (pRotMatrix.c2 - pRotMatrix.b3) / s;
116 y = (pRotMatrix.a3 - pRotMatrix.c1) / s;
117 z = (pRotMatrix.b1 - pRotMatrix.a2) / s;
118 w = static_cast<TReal>(0.25) * s;
119 } // else we have to check several cases
120 else if( pRotMatrix.a1 > pRotMatrix.b2 && pRotMatrix.a1 > pRotMatrix.c3 )
121 {
122 // Column 0:
123 TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.a1 - pRotMatrix.b2 - pRotMatrix.c3) * static_cast<TReal>(2.0);
124 x = static_cast<TReal>(0.25) * s;
125 y = (pRotMatrix.b1 + pRotMatrix.a2) / s;
126 z = (pRotMatrix.a3 + pRotMatrix.c1) / s;
127 w = (pRotMatrix.c2 - pRotMatrix.b3) / s;
128 }
129 else if( pRotMatrix.b2 > pRotMatrix.c3)
130 {
131 // Column 1:
132 TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.b2 - pRotMatrix.a1 - pRotMatrix.c3) * static_cast<TReal>(2.0);
133 x = (pRotMatrix.b1 + pRotMatrix.a2) / s;
134 y = static_cast<TReal>(0.25) * s;
135 z = (pRotMatrix.c2 + pRotMatrix.b3) / s;
136 w = (pRotMatrix.a3 - pRotMatrix.c1) / s;
137 } else
138 {
139 // Column 2:
140 TReal s = std::sqrt( static_cast<TReal>(1.0) + pRotMatrix.c3 - pRotMatrix.a1 - pRotMatrix.b2) * static_cast<TReal>(2.0);
141 x = (pRotMatrix.a3 + pRotMatrix.c1) / s;
142 y = (pRotMatrix.c2 + pRotMatrix.b3) / s;
143 z = static_cast<TReal>(0.25) * s;
144 w = (pRotMatrix.b1 - pRotMatrix.a2) / s;
145 }
146}

148// ---------------------------------------------------------------------------
149// Construction from euler angles
150template<typename TReal>
151inline aiQuaterniont<TReal>::aiQuaterniont( TReal fPitch, TReal fYaw, TReal fRoll )
152{
153 const TReal fSinPitch(std::sin(fPitch*static_cast<TReal>(0.5)));
154 const TReal fCosPitch(std::cos(fPitch*static_cast<TReal>(0.5)));
155 const TReal fSinYaw(std::sin(fYaw*static_cast<TReal>(0.5)));
156 const TReal fCosYaw(std::cos(fYaw*static_cast<TReal>(0.5)));
157 const TReal fSinRoll(std::sin(fRoll*static_cast<TReal>(0.5)));
158 const TReal fCosRoll(std::cos(fRoll*static_cast<TReal>(0.5)));
159 const TReal fCosPitchCosYaw(fCosPitch*fCosYaw);
160 const TReal fSinPitchSinYaw(fSinPitch*fSinYaw);
161 x = fSinRoll * fCosPitchCosYaw - fCosRoll * fSinPitchSinYaw;
162 y = fCosRoll * fSinPitch * fCosYaw + fSinRoll * fCosPitch * fSinYaw;
163 z = fCosRoll * fCosPitch * fSinYaw - fSinRoll * fSinPitch * fCosYaw;
164 w = fCosRoll * fCosPitchCosYaw + fSinRoll * fSinPitchSinYaw;
165}

167// ---------------------------------------------------------------------------
168// Returns a matrix representation of the quaternion
169template<typename TReal>
170inline aiMatrix3x3t<TReal> aiQuaterniont<TReal>::GetMatrix() const
171{
172 aiMatrix3x3t<TReal> resMatrix;
173 resMatrix.a1 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (y * y + z * z);
174 resMatrix.a2 = static_cast<TReal>(2.0) * (x * y - z * w);
175 resMatrix.a3 = static_cast<TReal>(2.0) * (x * z + y * w);
176 resMatrix.b1 = static_cast<TReal>(2.0) * (x * y + z * w);
177 resMatrix.b2 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (x * x + z * z);
178 resMatrix.b3 = static_cast<TReal>(2.0) * (y * z - x * w);
179 resMatrix.c1 = static_cast<TReal>(2.0) * (x * z - y * w);
180 resMatrix.c2 = static_cast<TReal>(2.0) * (y * z + x * w);
181 resMatrix.c3 = static_cast<TReal>(1.0) - static_cast<TReal>(2.0) * (x * x + y * y);
182
183 return resMatrix;
184}

186// ---------------------------------------------------------------------------
187// Construction from an axis-angle pair
188template<typename TReal>
189inline aiQuaterniont<TReal>::aiQuaterniont( aiVector3t<TReal> axis, TReal angle)
190{
191 axis.Normalize();
192
193 const TReal sin_a = std::sin( angle / 2 );
194 const TReal cos_a = std::cos( angle / 2 );
195 x = axis.x * sin_a;
196 y = axis.y * sin_a;
197 z = axis.z * sin_a;
198 w = cos_a;
199}
200// ---------------------------------------------------------------------------
201// Construction from am existing, normalized quaternion
202template<typename TReal>
203inline aiQuaterniont<TReal>::aiQuaterniont( aiVector3t<TReal> normalized)
204{
205 x = normalized.x;
206 y = normalized.y;
207 z = normalized.z;
208
209 const TReal t = static_cast<TReal>(1.0) - (x*x) - (y*y) - (z*z);
210
211 if (t < static_cast<TReal>(0.0)) {
212 w = static_cast<TReal>(0.0);
213 }
214 else w = std::sqrt (t);
215}

217// ---------------------------------------------------------------------------
218// Performs a spherical interpolation between two quaternions
219// Implementation adopted from the gmtl project. All others I found on the net fail in some cases.
220// Congrats, gmtl!
221template<typename TReal>
222inline void aiQuaterniont<TReal>::Interpolate( aiQuaterniont& pOut, const aiQuaterniont& pStart, const aiQuaterniont& pEnd, TReal pFactor)
223{
224 // calc cosine theta
225 TReal cosom = pStart.x * pEnd.x + pStart.y * pEnd.y + pStart.z * pEnd.z + pStart.w * pEnd.w;
226
227 // adjust signs (if necessary)
228 aiQuaterniont end = pEnd;
229 if( cosom < static_cast<TReal>(0.0))
230 {
231 cosom = -cosom;
232 end.x = -end.x; // Reverse all signs
233 end.y = -end.y;
234 end.z = -end.z;
235 end.w = -end.w;
236 }
237
238 // Calculate coefficients
239 TReal sclp, sclq;
240
241 if ((static_cast<TReal>(1.0) - cosom) > ai_epsilon) // 0.0001 -> some epsillon
242 {
243 // Standard case (slerp)
244 TReal omega, sinom;
245 omega = std::acos( cosom); // extract theta from dot product's cos theta
246 sinom = std::sin( omega);
247 sclp = std::sin( (static_cast<TReal>(1.0) - pFactor) * omega) / sinom;
248 sclq = std::sin( pFactor * omega) / sinom;
249 } else
250 {
251 // Very close, do linear interp (because it's faster)
252 sclp = static_cast<TReal>(1.0) - pFactor;
253 sclq = pFactor;
254 }
255
256 pOut.x = sclp * pStart.x + sclq * end.x;
257 pOut.y = sclp * pStart.y + sclq * end.y;
258 pOut.z = sclp * pStart.z + sclq * end.z;
259 pOut.w = sclp * pStart.w + sclq * end.w;
260}

263template<typename TReal>
264inline aiQuaterniont<TReal>& aiQuaterniont<TReal>::Normalize()
265{
266 // compute the magnitude and divide through it
267 const TReal mag = std::sqrt(x*x + y*y + z*z + w*w);
268 if (mag)
269 {
270 const TReal invMag = static_cast<TReal>(1.0)/mag;
271 x *= invMag;
272 y *= invMag;
273 z *= invMag;
274 w *= invMag;
275 }
276 return *this;
277}

282{
283 return aiQuaterniont(w*t.w - x*t.x - y*t.y - z*t.z,
284 w*t.x + x*t.w + y*t.z - z*t.y,
285 w*t.y + y*t.w + z*t.x - x*t.z,
286 w*t.z + z*t.w + x*t.y - y*t.x);

300template<typename TReal>
301inline aiVector3t<TReal> aiQuaterniont<TReal>::Rotate (const aiVector3t<TReal>& v) const
302{
303 aiQuaterniont q2(0.f,v.x,v.y,v.z), q = *this, qinv = q;
304 qinv.Conjugate();
305
306 q = q*q2*qinv;
307 return aiVector3t<TReal>(q.x,q.y,q.z);
