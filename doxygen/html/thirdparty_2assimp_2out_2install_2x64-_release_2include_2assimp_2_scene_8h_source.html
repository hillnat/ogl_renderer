418 std::pair<const aiTexture*, int> GetEmbeddedTextureAndIndex(const char* filename) const {
419 if (nullptr==filename) {
420 return std::make_pair(nullptr, -1);
421 }
422 // lookup using texture ID (if referenced like: "*1", "*2", etc.)
423 if ('*' == *filename) {
424 int index = std::atoi(filename + 1);
425 if (0 > index || mNumTextures <= static_cast<unsigned>(index)) {
426 return std::make_pair(nullptr, -1);
427 }
428 return std::make_pair(mTextures[index], index);
429 }
430 // lookup using filename
431 const char* shortFilename = GetShortFilename(filename);
432 if (nullptr == shortFilename) {
433 return std::make_pair(nullptr, -1);
434 }
435
436 for (unsigned int i = 0; i < mNumTextures; i++) {
437 const char* shortTextureFilename = GetShortFilename(mTextures[i]->mFilename.C_Str());
438 if (strcmp(shortTextureFilename, shortFilename) == 0) {
439 return std::make_pair(mTextures[i], static_cast<int>(i));
440 }
441 }
442 return std::make_pair(nullptr, -1);
443 }
