143template <class T>
144AI_FORCE_INLINE bool IOStreamBuffer<T>::open(IOStream *stream) {
145 // file still opened!
146 if (nullptr != m_stream) {
147 return false;
148 }
149
150 // Invalid stream pointer
151 if (nullptr == stream) {
152 return false;
153 }
154
155 m_stream = stream;
156 m_filesize = m_stream->FileSize();
157 if (m_filesize == 0) {
158 return false;
159 }
160 if (m_filesize < m_cacheSize) {
161 m_cacheSize = m_filesize;
162 }
163
164 m_numBlocks = m_filesize / m_cacheSize;
165 if ((m_filesize % m_cacheSize) > 0) {
166 m_numBlocks++;
167 }
168
169 return true;
170}

203template <class T>
204AI_FORCE_INLINE bool IOStreamBuffer<T>::readNextBlock() {
205 m_stream->Seek(m_filePos, aiOrigin_SET);
206 size_t readLen = m_stream->Read(&m_cache[0], sizeof(T), m_cacheSize);
207 if (readLen == 0) {
208 return false;
209 }
210 if (readLen < m_cacheSize) {
211 m_cacheSize = readLen;
212 }
213 m_filePos += m_cacheSize;
214 m_cachePos = 0;
215 m_blockIdx++;
216
217 return true;
218}

235template <class T>
236AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextDataLine(std::vector<T> &buffer, T continuationToken) {
237 buffer.resize(m_cacheSize);
238 if (m_cachePos >= m_cacheSize || 0 == m_filePos) {
239 if (!readNextBlock()) {
240 return false;
241 }
242 }
243
244 size_t i = 0;
245 for (;;) {
246 if (continuationToken == m_cache[m_cachePos] && IsLineEnd(m_cache[m_cachePos + 1])) {
247 ++m_cachePos;
248 while (m_cache[m_cachePos] != '\n') {
249 ++m_cachePos;
250 }
251 ++m_cachePos;
252 } else if (IsLineEnd(m_cache[m_cachePos])) {
253 break;
254 }
255
256 buffer[i] = m_cache[m_cachePos];
257 ++m_cachePos;
258 ++i;
259
260 if(i == buffer.size()) {
261 buffer.resize(buffer.size() * 2);
262 }
263
264 if (m_cachePos >= size()) {
265 break;
266 }
267 if (m_cachePos >= m_cacheSize) {
268 if (!readNextBlock()) {
269 return false;
270 }
271 }
272 }
273
274 buffer[i] = '\n';
275 ++m_cachePos;
276
277 return true;
278}

284template <class T>
285AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextLine(std::vector<T> &buffer) {
286 buffer.resize(m_cacheSize);
287 if (m_cachePos >= m_cacheSize || 0 == m_filePos) {
288 if (!readNextBlock()) {
289 return false;
290 }
291 }
292
293 if (IsLineEnd(m_cache[m_cachePos])) {
294 // skip line end
295 while (m_cache[m_cachePos] != '\n') {
296 ++m_cachePos;
297 }
298 ++m_cachePos;
299 if (isEndOfCache(m_cachePos, m_cacheSize)) {
300 if (!readNextBlock()) {
301 return false;
302 }
303 }
304 }
305
306 size_t i(0);
307 while (!IsLineEnd(m_cache[m_cachePos])) {
308 buffer[i] = m_cache[m_cachePos];
309 ++m_cachePos;
310 ++i;
311
312 if(i == buffer.size()) {
313 buffer.resize(buffer.size() * 2);
314 }
315
316 if (m_cachePos >= m_cacheSize) {
317 if (!readNextBlock()) {
318 return false;
319 }
320 }
321 }
322 buffer[i] = '\n';
323 while (m_cachePos < m_cacheSize && (m_cache[m_cachePos] == '\r' || m_cache[m_cachePos] == '\n')) {
324 ++m_cachePos;
325 }
326
327 return true;
328}

330template <class T>
331AI_FORCE_INLINE bool IOStreamBuffer<T>::getNextBlock(std::vector<T> &buffer) {
332 // Return the last block-value if getNextLine was used before
333 if (0 != m_cachePos) {
334 buffer = std::vector<T>(m_cache.begin() + m_cachePos, m_cache.end());
335 m_cachePos = 0;
336 } else {
337 if (!readNextBlock()) {
338 return false;
339 }
340
341 buffer = std::vector<T>(m_cache.begin(), m_cache.end());
342 }
343
344 return true;
345}
