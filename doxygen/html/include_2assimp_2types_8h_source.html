267struct aiString {
268#ifdef __cplusplus
270 aiString() AI_NO_EXCEPT :
271 length(0), data{'\0'} {
272#ifdef ASSIMP_BUILD_DEBUG
273 // Debug build: overwrite the string on its full length with ESC (27)
274 memset(data + 1, 27, AI_MAXLEN - 1);
275#endif
276 }
277
279 aiString(const aiString &rOther) :
280 length(rOther.length), data{'\0'} {
281 // Crop the string to the maximum length
282 length = length >= AI_MAXLEN ? AI_MAXLEN - 1 : length;
283 memcpy(data, rOther.data, length);
284 data[length] = '\0';
285 }
286
288 explicit aiString(const std::string &pString) :
289 length((ai_uint32)pString.length()), data{'\0'} {
290 length = length >= AI_MAXLEN ? AI_MAXLEN - 1 : length;
291 memcpy(data, pString.c_str(), length);
292 data[length] = '\0';
293 }
294
296 void Set(const std::string &pString) {
297 if (pString.length() > AI_MAXLEN - 1) {
298 return;
299 }
300 length = (ai_uint32)pString.length();
301 memcpy(data, pString.c_str(), length);
302 data[length] = 0;
303 }
304
306 void Set(const char *sz) {
307 ai_int32 len = (ai_uint32)::strlen(sz);
308 if (len > static_cast<ai_int32>(AI_MAXLEN - 1)) {
309 len = static_cast<ai_int32>(AI_MAXLEN - 1);
310 }
311 length = len;
312 memcpy(data, sz, len);
313 data[len] = 0;
314 }
315
317 aiString &operator=(const aiString &rOther) {
318 if (this == &rOther) {
319 return *this;
320 }
321
322 length = rOther.length;
323 if (length > (AI_MAXLEN - 1)) {
324 length = static_cast<ai_int32>(AI_MAXLEN - 1);
325 }
326
327 memcpy(data, rOther.data, length);
328 data[length] = '\0';
329 return *this;
330 }
331
333 aiString &operator=(const char *sz) {
334 Set(sz);
335 return *this;
336 }
337
339 aiString &operator=(const std::string &pString) {
340 Set(pString);
341 return *this;
342 }
343
345 bool operator==(const aiString &other) const {
346 if (length == other.length) {
347 return memcmp(data, other.data, length) == 0;
348 }
349 return false;
350 }
351
353 bool operator!=(const aiString &other) const {
354 return !(*this == other);
355 }
356
358 void Append(const char *app) {
359 const ai_uint32 len = static_cast<ai_uint32>(::strlen(app));
360 if (!len) {
361 return;
362 }
363 if (length + len >= AI_MAXLEN) {
364 return;
365 }
366
367 memcpy(&data[length], app, len + 1);
368 length += len;
369 }
370
372 void Clear() {
373 length = 0;
374 data[0] = '\0';
375
376#ifdef ASSIMP_BUILD_DEBUG
377 // Debug build: overwrite the string on its full length with ESC (27)
378 memset(data + 1, 27, AI_MAXLEN - 1);
379#endif
380 }
381
383 const char *C_Str() const {
384 return data;
385 }
386
387#endif // !__cplusplus
388
392 ai_uint32 length;
393
395 char data[AI_MAXLEN];
396}; // !struct aiString
