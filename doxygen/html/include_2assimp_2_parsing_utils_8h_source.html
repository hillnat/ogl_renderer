62// NOTE: the functions below are mostly intended as replacement for
63// std::upper, std::lower, std::isupper, std::islower, std::isspace.
64// we don't bother of locales. We don't want them. We want reliable
65// (i.e. identical) results across all locales.
66
67// The functions below accept any character type, but know only
68// about ASCII. However, UTF-32 is the only safe ASCII superset to
69// use since it doesn't have multi-byte sequences.

104template <class char_t>
105AI_FORCE_INLINE bool SkipSpaces(const char_t *in, const char_t **out, const char_t *end) {
106 while ((*in == (char_t)' ' || *in == (char_t)'\t') && in != end) {
107 ++in;
108 }
109 *out = in;
110 return !IsLineEnd<char_t>(*in);
111}

119// ---------------------------------------------------------------------------------
120template <class char_t>
121AI_FORCE_INLINE bool SkipLine(const char_t *in, const char_t **out, const char_t *end) {
122 while ((*in != (char_t)'\r' && *in != (char_t)'\n' && *in != (char_t)'\0') && in != end) {
123 ++in;
124 }
125
126 // files are opened in binary mode. Ergo there are both NL and CR
127 while ((*in == (char_t)'\r' || *in == (char_t)'\n') && in != end) {
128 ++in;
129 }
130 *out = in;
131 return *in != (char_t)'\0';
132}

156// ---------------------------------------------------------------------------------
157template <class char_t>
158AI_FORCE_INLINE bool GetNextLine(const char_t *&buffer, char_t out[BufferSize]) {
159 if ((char_t)'\0' == *buffer) {
160 return false;
161 }
162
163 char *_out = out;
164 char *const end = _out + BufferSize;
165 while (!IsLineEnd(*buffer) && _out < end) {
166 *_out++ = *buffer++;
167 }
168 *_out = (char_t)'\0';
169
170 while (IsLineEnd(*buffer) && '\0' != *buffer) {
171 ++buffer;
172 }
173
174 return true;
175}

184template <class char_t>
185AI_FORCE_INLINE bool TokenMatch(char_t *&in, const char *token, unsigned int len) {
186 if (!::strncmp(token, in, len) && IsSpaceOrNewLine(in[len])) {
187 if (in[len] != '\0') {
188 in += len + 1;
189 } else {
190 // If EOF after the token make sure we don't go past end of buffer
191 in += len;
192 }
193 return true;
194 }
195
196 return false;
197}

206 in += len + 1;

221AI_FORCE_INLINE std::string GetNextToken(const char *&in, const char *end) {
222 SkipSpacesAndLineEnd(&in, end);
223 const char *cur = in;
224 while (!IsSpaceOrNewLine(*in)) {
225 ++in;

237template <class string_type>
238AI_FORCE_INLINE unsigned int tokenize(const string_type &str, std::vector<string_type> &tokens,
239 const string_type &delimiters) {
240 // Skip delimiters at beginning.
241 typename string_type::size_type lastPos = str.find_first_not_of(delimiters, 0);
242
243 // Find first "non-delimiter".
244 typename string_type::size_type pos = str.find_first_of(delimiters, lastPos);
245 while (string_type::npos != pos || string_type::npos != lastPos) {
246 // Found a token, add it to the vector.
247 string_type tmp = str.substr(lastPos, pos - lastPos);
248 if (!tmp.empty() && ' ' != tmp[0])
249 tokens.push_back(tmp);
250
251 // Skip delimiters. Note the "not_of"
252 lastPos = str.find_first_not_of(delimiters, pos);
253
254 // Find next "non-delimiter"
255 pos = str.find_first_of(delimiters, lastPos);
256 }
257
258 return static_cast<unsigned int>(tokens.size());
259}
