175Texture MakeTexture(unsigned width, unsigned height, unsigned channels, const unsigned char* pixels) {
176 GLenum oglFormat = GL_RGBA;
177 switch (channels) {
178 case 1:
179 oglFormat = GL_RED;
180 break;
181 case 2:
182 oglFormat = GL_RG;
183 break;
184 case 3:
185 oglFormat = GL_RGB;
186 break;
187 default:
188 oglFormat = GL_RGBA;
189 break;
190 }
191 Texture textObj = { 0, width, height, channels };
192 glGenTextures(1, &textObj.handle);
193 //bind and buffer texture
194 glBindTexture(GL_TEXTURE_2D, textObj.handle);
195 glTexImage2D(GL_TEXTURE_2D, 0, oglFormat, width, height, 0, oglFormat, GL_UNSIGNED_BYTE, pixels);
196 //Configure texture settings
197 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
198 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
199 //Unbind and return the object
200 glBindTexture(GL_TEXTURE_2D, 0);
201 return textObj;
202}
